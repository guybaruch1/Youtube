import { useNavigate } from 'react-router-dom';
import './AuthBox.css';
import React, { useState } from 'react';

function AuthBox() {
  const [userName, setUserName] = useState('');
  const [password, setPassword] = useState('');
  const [isUserNameValid, setIsUserNameValid] = useState(false);
  const [isPasswordValid, setIsPasswordValid] = useState(false);

  // Function to handle changes in the userName input
  const handleUserNameChange = (e) => {
    setUserName(e.target.value);
    setIsUserNameValid(e.target.checkValidity());
  };

  // Function to handle changes in the password input
  const handlePasswordChange = (e) => {
    setPassword(e.target.value);
    setIsPasswordValid(e.target.checkValidity());
  };

  // Function to handle form submission
  const handleSubmit = (e) => {
    e.preventDefault();

    // Validation logic for userName
    if (!/^(?=.*[a-zA-Z])(?=.*[0-9])/.test(userName)) {
      alert('Please enter a valid userName containing both letters and numbers.');
      return;
    }

    // Validation logic for password
    if (password.length < 8) {
      alert('Password must be at least 8 characters long.');
      return;
    }

  };

  return (
    <div className="position-absolute top-50 start-50 translate-middle">
      <div className='container'>
        <div className="card">
          <div className="card-body">
            <div className="container overflow-hidden text-center">
              <div className="row gy-5">
                <div className="col-6">
                  <div className="p-3 text-start">
                    <h5 className="card-title">sign in / register</h5>
                    <p className="card-text">to continue to youtube</p>
                  </div>
                </div>
                <div className="col-6">
                  <div className="p-3">
                    <form onSubmit={handleSubmit}>
                      <div className="form-floating mb-3">

                        {/* userName validation */}
                        <input type="userName"
                          className="form-control"
                          id="userName"
                          placeholder="name@example.com"
                          pattern="[a-z0-9._%+-]+@[a-z0-9.-]+\.[a-z]{2,4}$"
                          value={userName} // Link with userName state
                          onChange={handleUserNameChange} // Call handleUserNameChange on change 
                        />
                        <div className="invalid-feedback">Please enter a valid userName.</div>
                        <label htmlFor="floatingInput">Email address</label>
                      </div>
                      <div className="form-floating">

                        {/* password validation */}
                        <input type="password"
                          className="form-control"
                          id="password"
                          placeholder="Password"
                          pattern=".{8,}"
                          required
                          value={password} // Link with password state
                          onChange={handlePasswordChange} // Call handlePasswordChange on change
                        />
                        <div className="invalid-feedback">Password must be at least 8 characters long.</div>
                        <label htmlFor="floatingPassword">Password</label>
                      </div>
                      <button type="submit" className="btn btn-primary" disabled={!isUserNameValid || !isPasswordValid}>
                        Login
                      </button>
                      </form>
                  </div>
                </div>
                <div className="col-6">
                  <div className="p-3">
                  </div>
                </div>
                <div className="col-6">
                  <div className="p-3">
                    <button type="button" className="btn btn-light me-5">Sign up</button>=hyt54rw
                  </div>
                </div>
              </div>
            </div>
          </div>
        </div>
      </div>
      {/* <script src="script.js"></script> */}
    </div>
  );
}

export default AuthBox;
